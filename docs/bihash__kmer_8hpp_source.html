182 auto dimer_valid = [&](size_t p) -> bool {
183 if (p + 1 > seq_len - 1)
184 return false;
185 char c1 = seq[p];
186 char c2 = seq[p + 1];
187 uint8_t a = BI_CONVERT_TAB[(unsigned char)c1];
188 uint8_t b = BI_CONVERT_TAB[(unsigned char)c2];
189 uint8_t loc = 10 * a + b;
190 return DIMER_TAB[loc] != 0;
191 };
192
193 if (even) {
194 if (pos >= seq_len - k)
195 return false;
196 if (!dimer_valid(pos + k - 1)) {
197 pos += k + 1;
198 // pos += k;
199 return init();
200 }
201 even = false;
202 ++pos;
203
204 extend_hashes(
205 fwd_hashes[1], rev_hashes[1], k, num_hashes, hash_arr.get());
206 return true;
207 }
208
209 if (pos >= seq_len - k)
210 return false;
211
212 if (!dimer_valid(pos + k - 1)) {
213 pos += k + 1;
214 return init();
215 }
216
217 bool can_roll_second = dimer_valid(pos + k);
218
219 for (int i = 0; i < 2; i++) {
220 if (i == 0 || can_roll_second) {
221
222 char out1 = seq[pos + i - 1];
223 char out2 = seq[pos + i + 1 - 1];
224
225 char in1 = seq[pos + k + i - 1];
226 char in2 = seq[pos + k + i + 1 - 1];
227
228 fwd_hashes[i] = next_forward_bi_hash(
229 fwd_hashes[i], k, out1, out2, in1, in2, DIMER_TAB, TAB_33R, TAB_31L);

233 }

237
238 ++pos;
239 even = true;

241 if (!can_roll_second) {
242 fwd_hashes[1] = 0;
243 rev_hashes[1] = 0;
244 }

303 bool roll_back()
304 {
305 if (!initialized) {
306 return init();
307 }
308
309 if (pos == 0) {
310 return false;
311 }
312
313 auto dimer_valid = [&](size_t p) -> bool {
314 if (p + 1 >= seq_len)
315 return false;
316 uint8_t a = BI_CONVERT_TAB[(unsigned char)seq[p]];
317 uint8_t b = BI_CONVERT_TAB[(unsigned char)seq[p + 1]];
318 uint8_t loc = 10 * a + b;
319 return DIMER_TAB[loc] != 0;
320 };
321
322 // Odd State
323 if (!even) {
324 extend_hashes(
325 fwd_hashes[0], rev_hashes[0], k, num_hashes, hash_arr.get());
326 --pos;
327 even = true;
328 return true;
329 }
330
331 // Even state
332 // First dimer (the one at pos-1, pos)
333 if (!dimer_valid(pos - 1)) {
334 if (pos <= k)
335 return false;
336 pos -= k;
337 return init();
338 }
339
340 // Check if the second rollback is possible
341 bool can_roll_second = true;
342 if (pos <= 1 || !dimer_valid(pos - 2)) {
343 can_roll_second = false;
344 }
345
346 // If the second hash is missing, restore it by reinitializing around pos-2
347 if (fwd_hashes[1] == 0) {
348 pos -= 2;
349 init();
350 pos++;
351 extend_hashes(
352 fwd_hashes[1], rev_hashes[1], k, num_hashes, hash_arr.get());
353 even = false;
354 return true;
355 }
356
357 // Perform 1 or 2 backward hash updates
358 for (int i = 0; i < 2; i++) {
359 if (i == 1 || can_roll_second) {
360
361 char out1 = seq[pos + k - 2 + i];
362 char out2 = seq[pos + k - 1 + i];
363 char in1 = seq[pos - 2 + i];
364 char in2 = seq[pos - 1 + i];
365

371 }
372 }
373
374 extend_hashes(fwd_hashes[1], rev_hashes[1], k, num_hashes, hash_arr.get());

377 even = false;
378 return true;

453 bool init()
454 {
455 // std::cerr << "running init" << std::endl;
456 even = true;
457 bool has_n = true;
458 while (pos <= seq_len - k + 1 && has_n) {
459 has_n = false;
460 for (unsigned i = 0; i < k && pos <= seq_len - k + 1; i += 2) {
461 const char c1 = seq[pos + k - i - 2];
462 const char c2 = seq[pos + k - i - 1];
463 const uint8_t a = BI_CONVERT_TAB[(unsigned char)c1];
464 const uint8_t b = BI_CONVERT_TAB[(unsigned char)c2];
465 const uint8_t loc = 10 * a + b;
466
467 if (DIMER_TAB[loc] == 0) {
468 if (b == 0) {
469 pos += k - i;
470 } else {
471 pos += k - i - 1;
472 }
473
474 has_n = true;
475 }
476 }
477 }
478
479 if (pos > seq_len - k) {
480 return false;
481 }

483 auto dimer_valid = [&](size_t p) -> bool {
484 if (p + 1 > seq_len - 1)
