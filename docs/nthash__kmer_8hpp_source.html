22inline uint64_t
23base_forward_hash(const char* seq,
24 unsigned k,
25 const uint64_t* primitive_tab = SEED_TAB,
26 const uint64_t* dimer_tab = DIMER_TAB,
27 const uint64_t* trimer_tab = TRIMER_TAB,
28 const uint64_t* tetramer_tab = TETRAMER_TAB,
29 const uint8_t* convert_tab = CONVERT_TAB)
30{
31 uint64_t h_val = 0;
32 for (unsigned i = 0; i < k - 3; i += 4) {
33 h_val = srol(h_val, 4);
34 uint8_t loc = 0;
35 loc += 64 * convert_tab[(unsigned char)seq[i]]; // NOLINT
36 loc += 16 * convert_tab[(unsigned char)seq[i + 1]]; // NOLINT
37 loc += 4 * convert_tab[(unsigned char)seq[i + 2]];
38 loc += convert_tab[(unsigned char)seq[i + 3]];
39 h_val ^= tetramer_tab[loc];
40 }
41 const unsigned remainder = k % 4;
42 if (remainder > 0) {
43 h_val = srol(h_val, remainder);
44 }
45 if (remainder == 3) {
46 uint8_t trimer_loc = 0;
47 trimer_loc += 16 * convert_tab[(unsigned char)seq[k - 3]]; // NOLINT
48 trimer_loc += 4 * convert_tab[(unsigned char)seq[k - 2]];
49 trimer_loc += convert_tab[(unsigned char)seq[k - 1]];
50 h_val ^= trimer_tab[trimer_loc];
51 } else if (remainder == 2) {
52 uint8_t dimer_loc = 0;
53 dimer_loc += 4 * convert_tab[(unsigned char)seq[k - 2]];
54 dimer_loc += convert_tab[(unsigned char)seq[k - 1]];
55 h_val ^= dimer_tab[dimer_loc];
56 } else if (remainder == 1) {
57 h_val ^= primitive_tab[(unsigned char)seq[k - 1]];
58 }
59 return h_val;
60}

116inline uint64_t
117base_reverse_hash(const char* seq,
118 unsigned k,
119 const uint64_t* primitive_tab = SEED_TAB,
120 const uint64_t* dimer_tab = DIMER_TAB,
121 const uint64_t* trimer_tab = TRIMER_TAB,
122 const uint64_t* tetramer_tab = TETRAMER_TAB,
123 const uint8_t* rc_convert_tab = RC_CONVERT_TAB)
124{
125 uint64_t h_val = 0;
126 const unsigned remainder = k % 4;
127 if (remainder == 3) {
128 uint8_t trimer_loc = 0;
129 trimer_loc += 16 * rc_convert_tab[(unsigned char)seq[k - 1]]; // NOLINT
130 trimer_loc += 4 * rc_convert_tab[(unsigned char)seq[k - 2]];
131 trimer_loc += rc_convert_tab[(unsigned char)seq[k - 3]];
132 h_val ^= trimer_tab[trimer_loc];
133 } else if (remainder == 2) {
134 uint8_t dimer_loc = 0;
135 dimer_loc += 4 * rc_convert_tab[(unsigned char)seq[k - 1]];
136 dimer_loc += rc_convert_tab[(unsigned char)seq[k - 2]];
137 h_val ^= dimer_tab[dimer_loc];
138 } else if (remainder == 1) {
139 h_val ^= primitive_tab[(unsigned char)seq[k - 1] & CP_OFF];
140 }
141 for (int i = (int)(k - remainder) - 1; i >= 3; i -= 4) {
142 h_val = srol(h_val, 4);
143 uint8_t loc = 0;
144 loc += 64 * rc_convert_tab[(unsigned char)seq[i]]; // NOLINT
145 loc += 16 * rc_convert_tab[(unsigned char)seq[i - 1]]; // NOLINT
146 loc += 4 * rc_convert_tab[(unsigned char)seq[i - 2]];
147 loc += rc_convert_tab[(unsigned char)seq[i - 3]];
148 h_val ^= tetramer_tab[loc];
149 }
150 return h_val;
151}

216inline void
217sub_hash(uint64_t fh_val,
218 uint64_t rh_val,
219 const char* kmer_seq,
220 const std::vector<unsigned>& positions,
221 const std::vector<unsigned char>& new_bases,
222 unsigned k,
223 unsigned m,
224 uint64_t* h_val,
225 const uint64_t* const* right_table = MS_TAB_33R,
226 const uint64_t* const* left_table = MS_TAB_31L)
227{
228 uint64_t b_val = 0;
229
230 for (size_t i = 0; i < positions.size(); i++) {
231 const auto pos = positions[i];
232 const auto new_base = new_bases[i];
233
234 fh_val ^= srol_table(
235 (unsigned char)kmer_seq[pos], k - 1 - pos, right_table, left_table);
236 fh_val ^= srol_table(new_base, k - 1 - pos, right_table, left_table);
237
238 rh_val ^= srol_table(
239 (unsigned char)kmer_seq[pos] & CP_OFF, pos, right_table, left_table);
240 rh_val ^= srol_table(new_base & CP_OFF, pos, right_table, left_table);
241 }
242
243 b_val = canonical(fh_val, rh_val);
244 extend_hashes(b_val, k, m, h_val);
245}
