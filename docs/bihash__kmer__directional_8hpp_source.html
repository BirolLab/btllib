448 bool init()
449 {
450 bool has_n = true;
451 while (pos <= seq_len - k + 1 && has_n) {
452 has_n = false;
453 for (unsigned i = 0; i < k && pos <= seq_len - k + 1; i++) {
454 if (SEED_TAB[(unsigned char)seq[pos + k - i - 1]] == SEED_N) {
455 pos += k - i;
